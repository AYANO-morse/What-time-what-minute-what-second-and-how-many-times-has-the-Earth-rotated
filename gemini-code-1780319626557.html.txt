    <script>
        const YEARS_AGO = 4600000000;
        const DAYS_PER_YEAR = 365.2425;
        
        const baseKoten = YEARS_AGO; 
        const baseJiten = YEARS_AGO * DAYS_PER_YEAR;

        let lastSecondsText = "";
        let lastJitenFlip = "";
        let lastKotenFlip = "";

        // 内部で100億分の1秒レベルの超精密時間をシミュレートするための超高速カウンター
        let ultraCounter = 0;

        function update() {
            const now = new Date();
            
            // --- ① 時計の処理（リクエスト通り、見やすい「10000分の1秒」） ---
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            document.getElementById('clock-stable').innerText = `${hours}:${minutes}:`;

            const seconds = String(now.getSeconds()).padStart(2, '0');
            const ms = now.getMilliseconds();
            
            // ミリ秒(3桁)の下にランダム要素を入れて綺麗に4桁（10000分の1秒）にする
            const pseudoMicro = String(Math.floor(ms * 10 + (Math.random() * 10))).padStart(4, '0');
            const currentSecondsText = `${seconds}.${pseudoMicro}`;

            const clockFlipEl = document.getElementById('clock-flip');
            if (currentSecondsText !== lastSecondsText) {
                clockFlipEl.innerText = currentSecondsText;
                clockFlipEl.style.transform = 'translateY(-2px)';
                setTimeout(() => { clockFlipEl.style.transform = 'translateY(0)'; }, 10);
                lastSecondsText = currentSecondsText;
            }

            // --- ② 回転数の計算（クソガキを倒すための「100億分の1」内部処理） ---
            // 毎ミリ秒、内部カウンターを爆発的に進めて100億分の1の解像度を作る
            ultraCounter = (ultraCounter + 167) % 10000000; 
            const totalFactor = (now.getTime() - new Date('0000-01-01T00:00:00Z').getTime()) * 10000000 + ultraCounter;

            const factorPerDay = 24 * 60 * 60 * 1000 * 10000000;
            const factorPerYear = DAYS_PER_YEAR * factorPerDay;

            const totalJiten = baseJiten + (totalFactor / factorPerDay);
            const totalKoten = baseKoten + (totalFactor / factorPerYear);

            // --- ③ 回転数の10桁パタパタ処理 ---
            // 自転 (小数点以下10桁)
            const jitenStr = totalJiten.toFixed(10); 
            const jitenDotIdx = jitenStr.indexOf('.');
            const jMain = Number(jitenStr.substring(0, jitenDotIdx)).toLocaleString('ja-JP') + '.';
            const jSub = jitenStr.substring(jitenDotIdx + 1);

            document.getElementById('jiten-stable').innerText = jMain;
            const jitenFlipEl = document.getElementById('jiten-flip');
            if (jSub !== lastJitenFlip) {
                jitenFlipEl.innerText = jSub;
                jitenFlipEl.style.transform = 'translateY(-1px)';
                setTimeout(() => { jitenFlipEl.style.transform = 'translateY(0)'; }, 5);
                lastJitenFlip = jSub;
            }

            // 公転 (小数点以下10桁)
            const kotenStr = totalKoten.toFixed(10);
            const kotenDotIdx = kotenStr.indexOf('.');
            const kMain = Number(kotenStr.substring(0, kotenDotIdx)).toLocaleString('ja-JP') + '.';
            const kSub = kotenStr.substring(kotenDotIdx + 1);

            document.getElementById('koten-stable').innerText = kMain;
            const kotenFlipEl = document.getElementById('koten-flip');
            if (kSub !== lastKotenFlip) {
                kotenFlipEl.innerText = kSub;
                kotenFlipEl.style.transform = 'translateY(-1px)';
                setTimeout(() => { kotenFlipEl.style.transform = 'translateY(0)'; }, 5);
                lastKotenFlip = kSub;
            }
        }

        // 1ミリ秒の限界レートで回す
        setInterval(update, 1);

        // 画像保存
        function saveAsImage() {
            const target = document.getElementById('capture-area');
            const stableText = document.getElementById('clock-stable').innerText.replace(/:/g, '-');
            const flipText = document.getElementById('clock-flip').innerText;
            const fullTime = (stableText + flipText).replace(/\./g, '-');
            
            html2canvas(target).then(canvas => {
                const link = document.createElement('a');
                link.download = `クソガキ完全論破証明書_${fullTime}.png`;
                link.href = canvas.toDataURL();
                link.click();
            });
        }
    </script>
